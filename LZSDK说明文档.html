苹果支付 LZIAPManager+handle.h

pp 需要实现 LZIAPManager+handle.h 中的两个方法 龙珠会主动调用

//pp获取产品列表给龙珠
+ (void)getLZProducts:(LZProducts _Nonnull )result;
//pp苹果支付成功后通知龙珠
+ (void)LZTradeWithID:(NSString *_Nonnull)orderID product:(LZIAPModel * _Nonnull)product complete:(LZTransactionCompleteHandle _Nonnull)complete;

demo

@implementation LZIAPManager (HandleImp)

+ (void)getLZProducts:(LZProducts _Nonnull )result {
    //获取产品列表
    LZIAPModel *model = [[LZIAPModel alloc] init];
    model.productID = @"666";
    model.cost = @(6);
    model.goods = @(6);
    
    LZIAPModel *model1 = [[LZIAPModel alloc] init];
    model1.productID = @"8888";
    model1.cost = @(60);
    model1.goods = @(60);
    
    if (result) {
        result(YES, @[model, model1]);
    }
    
}

+ (void)LZTradeWithID:(NSString *_Nonnull)orderID product:(LZIAPModel * _Nonnull)product complete:(LZTransactionCompleteHandle _Nonnull)complete {
    //支付
    //支付成功
    if (complete) {
        complete(YES, product, orderID, @"苹果返回的校验码");
    }
}

@end

LZDataReportManager 数据打点类

说明:一下是三种上报策略

    REPORT_SEND_REALTIME = 1,       //实时（5s发送一次）     (只在“集成测试”设备的DEBUG模式下有效)
    REPORT_SEND_LAUNCH = 2,         //启动发送
    REPORT_SEND_INTERVAL = 3,       //最小间隔发送  ([90-86400]s, default 90s)

说明：当 APP 启动后要调用此方法完成数据打点的配置

/**
 数据上报初始化配置

 @param policy 策略 默认为 REPORT_SEND_INTERVAL
 @param appkey  appkey string（服务器分配，不能为空）
 @param appscret appSecret string(服务器分配，不能为空)
 @param routeName route string 路由名称（服务器分配，不能为空）
 */
+ (void)startWithReportPolicy:(eReportPolicy)policy
                             withAppKey:(NSString *)appkey
                           withAppscret:(NSString *)appscret
                          withRouteName:(NSString *)routeName;

demo

- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions {
    [LZDataReportManager startWithReportPolicy:REPORT_SEND_REALTIME
                                    withAppKey:@"12123"
                                  withAppscret:@"1111"
                                 withRouteName:@"abc"];
    
    return YES;
}
